def kmeans_onevariable(var1,k_value):
    for var_i in data.describe().T.index:
        if var_i != var1:
            fig = plt.figure(figsize=(5, 4))
            est = KMeans(n_clusters=k_value)
            est.fit(data[[var1,var_i]])
            labels = est.labels_
            colors_labels = labels.astype(np.float)
            plt.scatter(data[var1],data[var_i], c=colors_labels)
            plt.title(var1+" vs. "+var_i)
